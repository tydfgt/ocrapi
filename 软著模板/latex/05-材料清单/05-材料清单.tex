\section{审核要点自查}

\subsection{一致性检查}
\begin{itemize}
    \item[\unchecked] 所有材料中\textbf{软件名称}完全一致（含大小写、标点、版本号）
    \item[\unchecked] 所有材料中\textbf{著作权人名称}一致
    \item[\unchecked] 开发完成日期在各文档中一致
    \item[\unchecked] 版本号格式统一（如 V1.0、V1.0.0）
\end{itemize}

\subsection{源代码检查}
\begin{itemize}
    \item[\unchecked] 源代码共 60 页
    \item[\unchecked] 前 30 页包含程序核心逻辑代码
    \item[\unchecked] 后 30 页为程序末尾代码
    \item[\unchecked] 每页不少于 50 行有效代码
    \item[\unchecked] 无第三方库、自动生成代码
    \item[\unchecked] 页眉标注软件名称 + 版本号
    \item[\unchecked] 页脚标注页码
\end{itemize}

\subsection{说明书检查}
\begin{itemize}
    \item[\unchecked] 包含软件运行截图（建议 8--10 张以上）
    \item[\unchecked] 每个核心功能均有操作说明
    \item[\unchecked] 包含运行环境说明
    \item[\unchecked] 包含安装/部署步骤
    \item[\unchecked] 总页数建议 15--30 页
\end{itemize}

\subsection{申请表检查}
\begin{itemize}
    \item[\unchecked] 所有必填项已填写完整
    \item[\unchecked] 软件分类号正确
    \item[\unchecked] 权利范围已勾选
    \item[\unchecked] 软件功能简述完整清晰（200--500 字）
\end{itemize}

\section{提交流程}

\begin{enumerate}
    \item 登录中国版权保护中心官网 \url{http://www.ccopyright.com.cn}
    \item 在线填写软件著作权登记申请表
    \item 打印申请表并签字/盖章
    \item 将所有纸质材料邮寄至版权保护中心
    \item 等待受理通知（约 1--2 周）
    \item 等待审查结果（约 2--4 个月）
    \item 领取著作权登记证书
\end{enumerate}
