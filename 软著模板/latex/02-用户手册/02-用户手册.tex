\section{软件概述}
\subsection{功能介绍}
PaddleOCR 智能文字识别系统是一套运行于 NVIDIA Jetson Orin Nano 边缘计算平台上的实时 OCR 服务软件，利用深度学习技术对上传图片进行文字检测与识别。

\textbf{核心功能}：图片文字识别、Web 可视化前端、RESTful API 服务、Base64 图片识别、硬件实时监控、GPU 预热加速。

\subsection{技术特点}
\begin{longtable}{|p{4cm}|p{10cm}|}\hline
边缘推理 & 全部计算在 Jetson 本地完成 \\\hline
GPU 加速 & CUDA 12.6 + cuDNN 9.3 并行计算 \\\hline
低延迟 & 预热后约 5.5 秒/图 \\\hline
轻量级 & PP-OCRv5 Mobile 模型（4.8MB+17MB） \\\hline
跨平台 API & 标准 RESTful 接口 \\\hline
\end{longtable}

\section{安装与部署}

\subsection{获取源代码}
\begin{lstlisting}[language=bash]
mkdir -p ~/ocrcplus && cd ~/ocrcplus
git clone --depth=1 https://gitee.com/paddlepaddle/PaddleOCR.git
\end{lstlisting}

\subsection{下载 Paddle Inference 推理库}
\begin{lstlisting}[language=bash]
cd ~/ocrcplus
wget "https://paddle-inference-lib.bj.bcebos.com/3.0.0/cxx_c/Jetson/jetpack5.1.2_gcc9.4/orin/paddle_inference_install_dir.tgz"
tar xzf paddle_inference_install_dir.tgz
\end{lstlisting}

\subsection{编译 C++ OCR 引擎}
\begin{lstlisting}[language=bash]
cd ~/ocrcplus/PaddleOCR/deploy/cpp_infer && mkdir build && cd build
cmake .. -DPADDLE_LIB=~/ocrcplus/paddle_inference_install_dir \
  -DOPENCV_DIR=/usr/local -DCUDA_LIB=/usr/local/cuda-12.6/lib64 \
  -DCUDNN_LIB=/usr/lib/aarch64-linux-gnu -DWITH_MKL=OFF -DWITH_GPU=ON
make -j2
\end{lstlisting}

\subsection{安装 Python 依赖}
\begin{lstlisting}[language=bash]
cd ~/ocrcplus && python3 -m venv venv && source venv/bin/activate
pip install fastapi uvicorn python-multipart Pillow psutil jetson-stats aiofiles aiohttp
\end{lstlisting}

\subsection{下载 OCR 模型}
\begin{lstlisting}[language=bash]
mkdir -p ~/ocrcplus/models && cd ~/ocrcplus/models
wget "https://paddle-model-ecology.bj.bcebos.com/paddlex/official_inference_model/paddle3.0.0/PP-OCRv5_mobile_det_infer.tar"
wget "https://paddle-model-ecology.bj.bcebos.com/paddlex/official_inference_model/paddle3.0.0/PP-OCRv5_mobile_rec_infer.tar"
tar xf PP-OCRv5_mobile_det_infer.tar && tar xf PP-OCRv5_mobile_rec_infer.tar
\end{lstlisting}

\subsection{启动服务}
\begin{lstlisting}[language=bash]
cd ~/ocrcplus && ./start.sh
# 输出: 🚀 PaddleOCR API 启动: http://0.0.0.0:8899
#       ✅ GPU 预热完成！后续请求 ~5.5s/图
\end{lstlisting}

\subsection{POST /ocr — 图片文字识别}
\begin{lstlisting}[language=bash]
curl -X POST http://192.168.3.110:8899/ocr -F "file=@test.png"
\end{lstlisting}

响应字段：texts（文字列表）、scores（置信度）、count（文本数）、vis\_base64（标注图片 Base64）。

\section{命令行使用}
\begin{lstlisting}[language=bash]
cd ~/ocrcplus
./PaddleOCR/deploy/cpp_infer/build/ppocr ocr \
  --input ./models/test.png \
  --text_detection_model_dir ./models/PP-OCRv5_mobile_det_infer \
  --text_detection_model_name PP-OCRv5_mobile_det \
  --text_recognition_model_dir ./models/PP-OCRv5_mobile_rec_infer \
  --text_recognition_model_name PP-OCRv5_mobile_rec \
  --use_doc_orientation_classify false \
  --use_doc_unwarping false --use_textline_orientation false \
  --device gpu --save_path ./output/
\end{lstlisting}

\section{常见问题}
\begin{enumerate}
\item \textbf{CPU 模式崩溃}：ARM 不支持 Intel MKL，使用 \texttt{--device gpu}。
\item \textbf{编译 OOM}：使用 \texttt{make -j2}，确保 Swap 已启用。
\item \textbf{模型名称不匹配}：确保 \texttt{--text\_detection\_model\_name} 与实际模型一致。
\end{enumerate}
